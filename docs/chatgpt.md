function fmt(n, digits=3){ return (n==null || Number.isNaN(n)) ? "—" : Number(n).toFixed(digits); }

function haversineKm(lat1, lon1, lat2, lon2){
  const R = 6371;
  const toRad = (d)=> d*Math.PI/180;
  const dLat = toRad(lat2-lat1);
  const dLon = toRad(lon2-lon1);
  const a =
    Math.sin(dLat/2)**2 +
    Math.cos(toRad(lat1))*Math.cos(toRad(lat2))*Math.sin(dLon/2)**2;
  return 2*R*Math.asin(Math.sqrt(a));
}

function bearingDeg(lat1, lon1, lat2, lon2){
  const toRad = (d)=> d*Math.PI/180;
  const toDeg = (r)=> r*180/Math.PI;
  const y = Math.sin(toRad(lon2-lon1)) * Math.cos(toRad(lat2));
  const x = Math.cos(toRad(lat1))*Math.sin(toRad(lat2)) -
            Math.sin(toRad(lat1))*Math.cos(toRad(lat2))*Math.cos(toRad(lon2-lon1));
  let brng = toDeg(Math.atan2(y, x));
  brng = (brng + 360) % 360;
  return brng;
}

function buildArcGISQueryURL(lat, lon, radiusKm){
  // ArcGIS REST query (FeatureServer/0)
  // We ask for points within radius of a point geometry.
  const params = new URLSearchParams({
    where: "1=1",
    outFields: "EMI_GUID,INCIDENTNAME,STATUS,TYPE,LOCATION,AGENCY,UPDATED,STARTED,INCIDENTNUM",
    geometry: `${lon},${lat}`,
    geometryType: "esriGeometryPoint",
    inSR: "4326",
    spatialRel: "esriSpatialRelIntersects",
    distance: String(radiusKm),
    units: "esriSRUnit_Kilometer",
    returnGeometry: "true",
    outSR: "4326",
    f: "pjson"
  });
  return `${FEEDS.incidentsArcGISLayer0}?${params.toString()}`;
}

function normaliseIncident(f){
  const a = f.attributes || {};
  const g = f.geometry || {};
  const lat = g.y ?? g.latitude ?? null;
  const lon = g.x ?? g.longitude ?? null;

  const updated = a.UPDATED ? new Date(a.UPDATED).toISOString() : null;
  const started = a.STARTED ? new Date(a.STARTED).toISOString() : null;

  return {
    id: a.EMI_GUID || a.INCIDENTNUM || a.INCIDENTNAME || Math.random().toString(16).slice(2),
    name: a.INCIDENTNAME || "Unnamed incident",
    status: a.STATUS || "Unknown",
    type: a.TYPE || "Unknown",
    location: a.LOCATION || "",
    agency: a.AGENCY || "",
    lat, lon,
    updatedISO: updated,
    startedISO: started
  };
}

function isFireType(incident, filter){
  if(filter==="all") return true;
  const t = (incident.type || "").toLowerCase();
  if(filter==="Fire") return t.includes("fire");
  return t.includes(filter.toLowerCase());
}

function statusLevel(statusText){
  const s = (statusText||"").toLowerCase();
  if(s.includes("safe") || s.includes("closed") || s.includes("contained") || s.includes("patrol")) return "good";
  if(s.includes("controlled") || s.includes("being controlled") || s.includes("under control")) return "warn";
  if(s.includes("out of control") || s.includes("going") || s.includes("not controlled") || s.includes("spreading")) return "bad";
  return "warn";
}

async function fetchRoutes(origin, dest){
  const url = `${FEEDS.osrmRoute}/${origin.lon},${origin.lat};${dest.lon},${dest.lat}?overview=full&geometries=geojson&alternatives=true&steps=false`;
  const res = await fetch(url, { headers: { "Accept":"application/json" }});
  if(!res.ok) throw new Error("Routing failed");
  const data = await res.json();
  if(data.code !== "Ok") throw new Error("Routing not OK");
  return (data.routes || []).map((r, idx)=> ({
    idx,
    distanceKm: r.distance/1000,
    durationMin: r.duration/60,
    geometry: r.geometry
  }));
}

function scoreRouteAgainstIncidents(route, incidents, bufferKm){
  const line = turf.lineString(route.geometry.coordinates);
  let near = [];
  for(const inc of incidents){
    const pt = turf.point([inc.lon, inc.lat]);
    const d = turf.pointToLineDistance(pt, line, { units: "kilometers" });
    if(d <= bufferKm){
      near.push({...inc, routeDistKm: d});
    }
  }
  near.sort((a,b)=> a.routeDistKm - b.routeDistKm);
  return { count: near.length, near };
}

function renderIncList(baseLat, baseLon, incidents, titlePrefix=""){
  const list = $("incList");
  list.innerHTML = "";

  const filter = $("typeFilter").value;
  const rows = incidents
    .filter(i => isFireType(i, filter))
    .map(i=>{
      const d = haversineKm(baseLat, baseLon, i.lat, i.lon);
      const b = bearingDeg(baseLat, baseLon, i.lat, i.lon);
      return {...i, distKm:d, bearing:b};
    })
    .sort((a,b)=> a.distKm - b.distKm)
    .slice(0, 60);

  if(!rows.length){
    list.innerHTML = `<div class="small">No incidents returned (or the feed is blocked). Use the official links on the left.</div>`;
    return;
  }

  for(const inc of rows){
    const lvl = statusLevel(inc.status);
    const vicEmLink = inc.id ? `https://emergency.vic.gov.au/respond/#/incident/${encodeURIComponent(inc.id)}` : "https://emergency.vic.gov.au/";
    const el = document.createElement("div");
    el.className = "item";
    el.innerHTML = `
      <div class="top">
        <div class="name">${escapeHtml(inc.name)}</div>
        <div class="badge ${lvl}">${escapeHtml(inc.status)}</div>
      </div>
      <div class="meta">
        <div><span class="muted">${escapeHtml(inc.type)}</span> • ${escapeHtml(inc.location || "")} • ${escapeHtml(inc.agency || "")}</div>

function setRiskFromNearest(labelEl, whyEl, nearest){
  if(!nearest){
    setStatus(labelEl, "warn", "Unknown (check VicEmergency)");
    whyEl.textContent = "No live incident data loaded. Use official links + radio.";
    return;
  }
  const lvl = statusLevel(nearest.status);
  if(nearest.distKm <= 30 && lvl==="bad"){
    setStatus(labelEl, "bad", "Higher (nearby active fire)");
    whyEl.textContent = `${nearest.name} is ~${nearest.distKm.toFixed(1)} km away (${nearest.type}, ${nearest.status}).`;
    return;
  }
  if(nearest.distKm <= 50 && (lvl==="bad" || lvl==="warn")){
    setStatus(labelEl, "warn", "Moderate (watch closely)");
    whyEl.textContent = `${nearest.name} is ~${nearest.distKm.toFixed(1)} km away (${nearest.type}, ${nearest.status}).`;
    return;
  }
  setStatus(labelEl, "good", "Lower (no close active fires found)");
  whyEl.textContent = nearest ? `Nearest: ${nearest.name} ~${nearest.distKm.toFixed(1)} km (${nearest.status}).` : "";

  if(state.routes.length && allForRoute.length){
    const scored = state.routes.map(r=>{
      const s = scoreRouteAgainstIncidents(r, allForRoute, DEFAULTS.routeBufferKm);
      return {...r, nearCount: s.count, near: s.near};
    }).sort((a,b)=> (a.nearCount - b.nearCount) || (a.durationMin - b.durationMin));

function nearestIncident(lat, lon, incidents){
  if(!incidents?.length) return null;
  const filter = $("typeFilter").value;
  let best = null;
  for(const i of incidents){
    if(!isFireType(i, filter)) continue;
    const d = haversineKm(lat, lon, i.lat, i.lon);
    const b = bearingDeg(lat, lon, i.lat, i.lon);
    const cand = {...i, distKm:d, bearing:b};
    if(!best || cand.distKm < best.distKm) best = cand;
  }
  return best;
}
